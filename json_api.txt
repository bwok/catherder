// ajax calls use the /api url


// api/updatemeetup
REQUEST:
{
    adminhash: string,              // hash. If set to null, a new meetup is created
	description: string,
	dates: [
		{
		    date: int,	            // signed 64 bit millisecond UNIX timestamp. Minimum value = 0.
			users: [
				{
				    name: string,
				    available: bool
				}, ....
			]
		}, ....
	],
	admin: {
		email: string,
		alerts: boolean
	}
}
RESPONSE:
{
    result: {
        userhash: string,           // hash
        adminhash: string           // hash
    },
    error: string                   // empty string when no error
}


// api/getusermeetup
REQUEST:
{
    userhash: string                // hash
}
RESPONSE:
{
	result: {
	    description: string,
        dates: [
            {
                date: int,	            // signed 64 bit millisecond UNIX timestamp
                users: [
                    {
                        name: string,
                        available: string
                    }, ....
                ]
            }, ....
        ]
    },
    error: string
}


// api/getadminmeetup

        dates: [

                        available: string
                    }, ....

// api/deletemeetup
REQUEST:
{
    adminhash: string               // hash
}
RESPONSE:
{
    result: string
    error: string                   // empty string when no error
}


// api/updateuser
REQUEST:
{
    userhash: string,               // hash
    username: string,               // If the username already exists, the existing user gets updated, else the user gets created.
    dates: [int, ....],	            // Dates the user is available for. Signed 64 bit millisecond UNIX timestamps
}
RESPONSE:
{
    result: string
    error: string                   // empty string when no error
}


// api/deleteuser
REQUEST:
{
    userhash: string,               // hash
    username: string
}
RESPONSE:
{
    result: string
    error: string                   // empty string when no error
}
